cat
  Abrir arquivos.
  cat arquivo

cd
  Mudar diretório.
  cd /diretorio

cowsay
  Programa de mensagens.
  cowsay -l

echo
  Diz o nome de uma variável.
  echo $HOSTNAME

env
  Lista as variáveis de ambiente.
  env

exit
  Desempilha os shells que foram criados.
  exit

help
  Lista comandos e suas funções. "help comando" especifica mais sobre o comando.
  help env
